4 * ----------------------------------------------------------------------------
5 * Example sketch/program which will try the most used default keys listed in
6 * https://code.google.com/p/mfcuk/wiki/MifareClassicDefaultKeys to dump the
7 * block 0 of a MIFARE RFID card using a RFID-RC522 reader.
8 *
9 * Typical pin layout used:
10 * -----------------------------------------------------------------------------------------
11 * MFRC522 Arduino Arduino Arduino Arduino Arduino
12 * Reader/PCD Uno/101 Mega Nano v3 Leonardo/Micro Pro Micro
13 * Signal Pin Pin Pin Pin Pin Pin
14 * -----------------------------------------------------------------------------------------
15 * RST/Reset RST 9 5 D9 RESET/ICSP-5 RST
16 * SPI SS SDA(SS) 10 53 D10 10 10
17 * SPI MOSI MOSI 11 / ICSP-4 51 D11 ICSP-4 16
18 * SPI MISO MISO 12 / ICSP-1 50 D12 ICSP-1 14
19 * SPI SCK SCK 13 / ICSP-3 52 D13 ICSP-3 15
20 *
21 * More pin layouts for other boards can be found here: https://github.com/miguelbalboa/rfid#pin-layout

91/*
92 * Try using the PICC (the tag/card) with the given key to access block 0 to 63.
93 * On success, it will show the key details, and dump the block data on Serial.
94 *
95 * @return true when the given key worked, false otherwise.
96 */
97
98bool try_key(MFRC522::MIFARE_Key *key)
99{
100 bool result = false;
101
102 for(byte block = 0; block < 64; block++){
103
104 // Serial.println(F("Authenticating using key A..."));
105 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, block, key, &(mfrc522.uid));
106 if (status != MFRC522::STATUS_OK) {
107 Serial.print(F("PCD_Authenticate() failed: "));
108 Serial.println(mfrc522.GetStatusCodeName(status));
109 return false;
110 }
111
112 // Read block
113 byte byteCount = sizeof(buffer);
114 status = mfrc522.MIFARE_Read(block, buffer, &byteCount);
115 if (status != MFRC522::STATUS_OK) {
116 Serial.print(F("MIFARE_Read() failed: "));
117 Serial.println(mfrc522.GetStatusCodeName(status));
118 }
119 else {
120 // Successful read
121 result = true;
122 Serial.print(F("Success with key:"));
123 dump_byte_array((*key).keyByte, MFRC522::MF_KEY_SIZE);
124 Serial.println();
125
126 // Dump block data
127 Serial.print(F("Block ")); Serial.print(block); Serial.print(F(":"));
128 dump_byte_array1(buffer, 16); //omzetten van hex naar ASCI
129 Serial.println();
130
131 for (int p = 0; p < 16; p++) //De 16 bits uit de block uitlezen
132 {
133 waarde [block][p] = buffer[p];
134 Serial.print(waarde[block][p]);
135 Serial.print(" ");
136 }
137
138 }
139 }
140 Serial.println();
141
142 Serial.println("1.Read card \n2.Write to card \n3.Copy the data.");
143
144 mfrc522.PICC_HaltA(); // Halt PICC
145 mfrc522.PCD_StopCrypto1(); // Stop encryption on PCD
146 return result;
147
148 start();
149}
