1/*
2 * ----------------------------------------------------------------------------
3 * This is a MFRC522 library example; see https://github.com/miguelbalboa/rfid
4 * for further details and other examples.
5 *
6 * NOTE: The library file MFRC522.h has a lot of useful info. Please read it.
7 *
8 * Released into the public domain.
9 * ----------------------------------------------------------------------------
10 * Example sketch/program which will try the most used default keys listed in
11 * https://code.google.com/p/mfcuk/wiki/MifareClassicDefaultKeys to dump the
12 * block 0 of a MIFARE RFID card using a RFID-RC522 reader.
13 *
14 * Typical pin layout used:
15 * -----------------------------------------------------------------------------------------
16 * MFRC522 Arduino Arduino Arduino Arduino Arduino
17 * Reader/PCD Uno/101 Mega Nano v3 Leonardo/Micro Pro Micro
18 * Signal Pin Pin Pin Pin Pin Pin
19 * -----------------------------------------------------------------------------------------
20 * RST/Reset RST 9 5 D9 RESET/ICSP-5 RST
21 * SPI SS SDA(SS) 10 53 D10 10 10
22 * SPI MOSI MOSI 11 / ICSP-4 51 D11 ICSP-4 16
23 * SPI MISO MISO 12 / ICSP-1 50 D12 ICSP-1 14
24 * SPI SCK SCK 13 / ICSP-3 52 D13 ICSP-3 15
25 *
26 * More pin layouts for other boards can be found here: https://github.com/miguelbalboa/rfid#pin-layout
27 *
28 */
29
30#include <SPI.h>
31#include <MFRC522.h>
32
33#define RST_PIN 9 // Configurable, see typical pin layout above
34#define SS_PIN 10 // Configurable, see typical pin layout above
35
36MFRC522 mfrc522(SS_PIN, RST_PIN); // Create MFRC522 instance.
37
38// Number of known default keys (hard-coded)
39// NOTE: Synchronize the NR_KNOWN_KEYS define with the defaultKeys[] array
40#define NR_KNOWN_KEYS 8
41// Known keys, see: https://code.google.com/p/mfcuk/wiki/MifareClassicDefaultKeys
42byte knownKeys[NR_KNOWN_KEYS][MFRC522::MF_KEY_SIZE] = {
43 {0xff, 0xff, 0xff, 0xff, 0xff, 0xff}, // FF FF FF FF FF FF = factory default
44 {0xa0, 0xa1, 0xa2, 0xa3, 0xa4, 0xa5}, // A0 A1 A2 A3 A4 A5
45 {0xb0, 0xb1, 0xb2, 0xb3, 0xb4, 0xb5}, // B0 B1 B2 B3 B4 B5
46 {0x4d, 0x3a, 0x99, 0xc3, 0x51, 0xdd}, // 4D 3A 99 C3 51 DD
47 {0x1a, 0x98, 0x2c, 0x7e, 0x45, 0x9a}, // 1A 98 2C 7E 45 9A
48 {0xd3, 0xf7, 0xd3, 0xf7, 0xd3, 0xf7}, // D3 F7 D3 F7 D3 F7
49 {0xaa, 0xbb, 0xcc, 0xdd, 0xee, 0xff}, // AA BB CC DD EE FF
50 {0x00, 0x00, 0x00, 0x00, 0x00, 0x00} // 00 00 00 00 00 00
51};

56void setup() {
57 Serial.begin(9600); // Initialize serial communications with the PC
58 while (!Serial); // Do nothing if no serial port is opened (added for Arduinos based on ATMEGA32U4)
59 SPI.begin(); // Init SPI bus
60 mfrc522.PCD_Init(); // Init MFRC522 card
61 Serial.println(F("Try the most used default keys to print block 0 of a MIFARE PICC."));
62}

64/*
65 * Helper routine to dump a byte array as hex values to Serial.
66 */
67void dump_byte_array(byte *buffer, byte bufferSize) {
68 for (byte i = 0; i < bufferSize; i++) {
69 Serial.print(buffer[i] < 0x10 ? " 0" : " ");
70 Serial.print(buffer[i], HEX);
71 }
72}

74/*
75 * Try using the PICC (the tag/card) with the given key to access block 0.
76 * On success, it will show the key details, and dump the block data on Serial.
77 *
78 * @return true when the given key worked, false otherwise.
79 */
80bool try_key(MFRC522::MIFARE_Key *key)
81{
82 bool result = false;
83 byte buffer[18];
84 byte block = 0;
85 MFRC522::StatusCode status;
86
87 // Serial.println(F("Authenticating using key A..."));
88 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, block, key, &(mfrc522.uid));
89 if (status != MFRC522::STATUS_OK) {
90 // Serial.print(F("PCD_Authenticate() failed: "));
91 // Serial.println(mfrc522.GetStatusCodeName(status));
92 return false;
93 }
94
95 // Read block
96 byte byteCount = sizeof(buffer);
97 status = mfrc522.MIFARE_Read(block, buffer, &byteCount);
98 if (status != MFRC522::STATUS_OK) {
99 // Serial.print(F("MIFARE_Read() failed: "));
100 // Serial.println(mfrc522.GetStatusCodeName(status));
101 }
102 else {
103 // Successful read
104 result = true;
105 Serial.print(F("Success with key:"));
106 dump_byte_array((*key).keyByte, MFRC522::MF_KEY_SIZE);
107 Serial.println();
108 // Dump block data
109 Serial.print(F("Block ")); Serial.print(block); Serial.print(F(":"));
110 dump_byte_array(buffer, 16);
111 Serial.println();
112 }
113 Serial.println();
114
115 mfrc522.PICC_HaltA(); // Halt PICC
116 mfrc522.PCD_StopCrypto1(); // Stop encryption on PCD
117 return result;
118}

120/*
121 * Main loop.
122 */
123void loop() {
124 // Reset the loop if no new card present on the sensor/reader. This saves the entire process when idle.
125 if ( ! mfrc522.PICC_IsNewCardPresent())
126 return;
127
128 // Select one of the cards
129 if ( ! mfrc522.PICC_ReadCardSerial())
130 return;
131
132 // Show some details of the PICC (that is: the tag/card)
133 Serial.print(F("Card UID:"));
134 dump_byte_array(mfrc522.uid.uidByte, mfrc522.uid.size);
135 Serial.println();
136 Serial.print(F("PICC type: "));
137 MFRC522::PICC_Type piccType = mfrc522.PICC_GetType(mfrc522.uid.sak);
138 Serial.println(mfrc522.PICC_GetTypeName(piccType));
139
140 // Try the known default keys
141 MFRC522::MIFARE_Key key;
142 for (byte k = 0; k < NR_KNOWN_KEYS; k++) {
143 // Copy the known key into the MIFARE_Key structure
144 for (byte i = 0; i < MFRC522::MF_KEY_SIZE; i++) {
145 key.keyByte[i] = knownKeys[k][i];
146 }
147 // Try the key
148 if (try_key(&key)) {
149 // Found and reported on the key and block,
150 // no need to try other keys for this PICC
151 break;
152 }
153
154 // http://arduino.stackexchange.com/a/14316
155 if ( ! mfrc522.PICC_IsNewCardPresent())
156 break;
157 if ( ! mfrc522.PICC_ReadCardSerial())
158 break;
159 }
160}
