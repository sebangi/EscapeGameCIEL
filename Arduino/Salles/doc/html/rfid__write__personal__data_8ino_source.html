1/*
2 * Write personal data of a MIFARE RFID card using a RFID-RC522 reader
3 * Uses MFRC522 - Library to use ARDUINO RFID MODULE KIT 13.56 MHZ WITH TAGS SPI W AND R BY COOQROBOT.
4 * -----------------------------------------------------------------------------------------
5 * MFRC522 Arduino Arduino Arduino Arduino Arduino
6 * Reader/PCD Uno/101 Mega Nano v3 Leonardo/Micro Pro Micro
7 * Signal Pin Pin Pin Pin Pin Pin
8 * -----------------------------------------------------------------------------------------
9 * RST/Reset RST 9 5 D9 RESET/ICSP-5 RST
10 * SPI SS SDA(SS) 10 53 D10 10 10
11 * SPI MOSI MOSI 11 / ICSP-4 51 D11 ICSP-4 16
12 * SPI MISO MISO 12 / ICSP-1 50 D12 ICSP-1 14
13 * SPI SCK SCK 13 / ICSP-3 52 D13 ICSP-3 15
14 *
15 * More pin layouts for other boards can be found here: https://github.com/miguelbalboa/rfid#pin-layout
16 *
17 * Hardware required:
18 * Arduino
19 * PCD (Proximity Coupling Device): NXP MFRC522 Contactless Reader IC
20 * PICC (Proximity Integrated Circuit Card): A card or tag using the ISO 14443A interface, eg Mifare or NTAG203.
21 * The reader can be found on eBay for around 5 dollars. Search for "mf-rc522" on ebay.com.
22 */

24#include <SPI.h>
25#include <MFRC522.h>
26
27#define RST_PIN 9 // Configurable, see typical pin layout above
28#define SS_PIN 10 // Configurable, see typical pin layout above
29
30MFRC522 mfrc522(SS_PIN, RST_PIN); // Create MFRC522 instance
31
32void setup() {
33 Serial.begin(9600); // Initialize serial communications with the PC
34 SPI.begin(); // Init SPI bus
35 mfrc522.PCD_Init(); // Init MFRC522 card
36 Serial.println(F("Write personal data on a MIFARE PICC "));
37}
38
39void loop() {
40
41 // Prepare key - all keys are set to FFFFFFFFFFFFh at chip delivery from the factory.
42 MFRC522::MIFARE_Key key;
43 for (byte i = 0; i < 6; i++) key.keyByte[i] = 0xFF;
44
45 // Reset the loop if no new card present on the sensor/reader. This saves the entire process when idle.
46 if ( ! mfrc522.PICC_IsNewCardPresent()) {
47 return;
48 }
49
50 // Select one of the cards
51 if ( ! mfrc522.PICC_ReadCardSerial()) {
52 return;
53 }
54
55 Serial.print(F("Card UID:")); //Dump UID
56 for (byte i = 0; i < mfrc522.uid.size; i++) {
57 Serial.print(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " ");
58 Serial.print(mfrc522.uid.uidByte[i], HEX);
59 }
60 Serial.print(F(" PICC type: ")); // Dump PICC type
61 MFRC522::PICC_Type piccType = mfrc522.PICC_GetType(mfrc522.uid.sak);
62 Serial.println(mfrc522.PICC_GetTypeName(piccType));
63
64 byte buffer[34];
65 byte block;
66 MFRC522::StatusCode status;
67 byte len;
68
69 Serial.setTimeout(20000L) ; // wait until 20 seconds for input from serial
70 // Ask personal data: Family name
71 Serial.println(F("Type Family name, ending with #"));
72 len = Serial.readBytesUntil('#', (char *) buffer, 30) ; // read family name from serial
73 for (byte i = len; i < 30; i++) buffer[i] = ' '; // pad with spaces
74
75 block = 1;
76 //Serial.println(F("Authenticating using key A..."));
77 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, block, &key, &(mfrc522.uid));
78 if (status != MFRC522::STATUS_OK) {
79 Serial.print(F("PCD_Authenticate() failed: "));
80 Serial.println(mfrc522.GetStatusCodeName(status));
81 return;
82 }
83 else Serial.println(F("PCD_Authenticate() success: "));
84
85 // Write block
86 status = mfrc522.MIFARE_Write(block, buffer, 16);
87 if (status != MFRC522::STATUS_OK) {
88 Serial.print(F("MIFARE_Write() failed: "));
89 Serial.println(mfrc522.GetStatusCodeName(status));
90 return;
91 }
92 else Serial.println(F("MIFARE_Write() success: "));
93
94 block = 2;
95 //Serial.println(F("Authenticating using key A..."));
96 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, block, &key, &(mfrc522.uid));
97 if (status != MFRC522::STATUS_OK) {
98 Serial.print(F("PCD_Authenticate() failed: "));
99 Serial.println(mfrc522.GetStatusCodeName(status));
100 return;
101 }
102
103 // Write block
104 status = mfrc522.MIFARE_Write(block, &buffer[16], 16);
105 if (status != MFRC522::STATUS_OK) {
106 Serial.print(F("MIFARE_Write() failed: "));
107 Serial.println(mfrc522.GetStatusCodeName(status));
108 return;
109 }
110 else Serial.println(F("MIFARE_Write() success: "));
111
112 // Ask personal data: First name
113 Serial.println(F("Type First name, ending with #"));
114 len = Serial.readBytesUntil('#', (char *) buffer, 20) ; // read first name from serial
115 for (byte i = len; i < 20; i++) buffer[i] = ' '; // pad with spaces
116
117 block = 4;
118 //Serial.println(F("Authenticating using key A..."));
119 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, block, &key, &(mfrc522.uid));
120 if (status != MFRC522::STATUS_OK) {
121 Serial.print(F("PCD_Authenticate() failed: "));
122 Serial.println(mfrc522.GetStatusCodeName(status));
123 return;
124 }
125
126 // Write block
127 status = mfrc522.MIFARE_Write(block, buffer, 16);
128 if (status != MFRC522::STATUS_OK) {
129 Serial.print(F("MIFARE_Write() failed: "));
130 Serial.println(mfrc522.GetStatusCodeName(status));
131 return;
132 }
133 else Serial.println(F("MIFARE_Write() success: "));
134
135 block = 5;
136 //Serial.println(F("Authenticating using key A..."));
137 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, block, &key, &(mfrc522.uid));
138 if (status != MFRC522::STATUS_OK) {
139 Serial.print(F("PCD_Authenticate() failed: "));
140 Serial.println(mfrc522.GetStatusCodeName(status));
141 return;
142 }
143
144 // Write block
145 status = mfrc522.MIFARE_Write(block, &buffer[16], 16);
146 if (status != MFRC522::STATUS_OK) {
147 Serial.print(F("MIFARE_Write() failed: "));
148 Serial.println(mfrc522.GetStatusCodeName(status));
149 return;
150 }
151 else Serial.println(F("MIFARE_Write() success: "));
152
153
154 Serial.println(" ");
155 mfrc522.PICC_HaltA(); // Halt PICC
156 mfrc522.PCD_StopCrypto1(); // Stop encryption on PCD
157
158}
