95void MFRC522::PCD_ReadRegister( PCD_Register reg,
96 byte count,
97 byte *values,
98 byte rxAlign
99 ) {
100 if (count == 0) {
101 return;
102 }
103 //Serial.print(F("Reading ")); Serial.print(count); Serial.println(F(" bytes from register."));
104 byte address = 0x80 | reg; // MSB == 1 is for reading. LSB is not used in address. Datasheet section 8.1.2.3.
105 byte index = 0; // Index in values array.
106 SPI.beginTransaction(SPISettings(MFRC522_SPICLOCK, MSBFIRST, SPI_MODE0)); // Set the settings to work with SPI bus
107 digitalWrite(_chipSelectPin, LOW); // Select slave
108 count--; // One read is performed outside of the loop
109 SPI.transfer(address); // Tell MFRC522 which address we want to read
110 if (rxAlign) { // Only update bit positions rxAlign..7 in values[0]
111 // Create bit mask for bit positions rxAlign..7
112 byte mask = (0xFF << rxAlign) & 0xFF;
113 // Read value and tell that we want to read the same address again.
114 byte value = SPI.transfer(address);
115 // Apply mask to both current value of values[0] and the new data in value.
116 values[0] = (values[0] & ~mask) | (value & mask);
117 index++;
118 }
119 while (index < count) {
120 values[index] = SPI.transfer(address); // Read value and tell that we want to read the same address again.
121 index++;
122 }
123 values[index] = SPI.transfer(0); // Read the final byte. Send 0 to stop reading.
124 digitalWrite(_chipSelectPin, HIGH); // Release slave again
125 SPI.endTransaction(); // Stop using the SPI bus
126} // End PCD_ReadRegister()

156MFRC522::StatusCode MFRC522::PCD_CalculateCRC( byte *data,
157 byte length,
158 byte *result
159 ) {
160 PCD_WriteRegister(CommandReg, PCD_Idle); // Stop any active command.
161 PCD_WriteRegister(DivIrqReg, 0x04); // Clear the CRCIRq interrupt request bit
162 PCD_WriteRegister(FIFOLevelReg, 0x80); // FlushBuffer = 1, FIFO initialization
163 PCD_WriteRegister(FIFODataReg, length, data); // Write data to the FIFO
164 PCD_WriteRegister(CommandReg, PCD_CalcCRC); // Start the calculation
165
166 // Wait for the CRC calculation to complete. Check for the register to
167 // indicate that the CRC calculation is complete in a loop. If the
168 // calculation is not indicated as complete in ~90ms, then time out
169 // the operation.
170 const uint32_t deadline = millis() + 89;
171
172 do {
173 // DivIrqReg[7..0] bits are: Set2 reserved reserved MfinActIRq reserved CRCIRq reserved reserved
174 byte n = PCD_ReadRegister(DivIrqReg);
175 if (n & 0x04) { // CRCIRq bit set - calculation done
176 PCD_WriteRegister(CommandReg, PCD_Idle); // Stop calculating CRC for new content in the FIFO.
177 // Transfer the result from the registers to the result buffer
178 result[0] = PCD_ReadRegister(CRCResultRegL);
179 result[1] = PCD_ReadRegister(CRCResultRegH);
180 return STATUS_OK;
181 }
182 yield();
183 }
184 while (static_cast<uint32_t> (millis()) < deadline);
185
186 // 89ms passed and nothing happened. Communication with the MFRC522 might be down.
187 return STATUS_TIMEOUT;
188} // End PCD_CalculateCRC()

198void MFRC522::PCD_Init() {
199 bool hardReset = false;
200
201 // Set the chipSelectPin as digital output, do not select the slave yet
202 pinMode(_chipSelectPin, OUTPUT);
203 digitalWrite(_chipSelectPin, HIGH);
204
205 // If a valid pin number has been set, pull device out of power down / reset state.
206 if (_resetPowerDownPin != UNUSED_PIN) {
207 // First set the resetPowerDownPin as digital input, to check the MFRC522 power down mode.
208 pinMode(_resetPowerDownPin, INPUT);
209
210 if (digitalRead(_resetPowerDownPin) == LOW) { // The MFRC522 chip is in power down mode.
211 pinMode(_resetPowerDownPin, OUTPUT); // Now set the resetPowerDownPin as digital output.
212 digitalWrite(_resetPowerDownPin, LOW); // Make sure we have a clean LOW state.
213 delayMicroseconds(2); // 8.8.1 Reset timing requirements says about 100ns. Let us be generous: 2μsl
214 digitalWrite(_resetPowerDownPin, HIGH); // Exit power down mode. This triggers a hard reset.
215 // Section 8.8.2 in the datasheet says the oscillator start-up time is the start up time of the crystal + 37,74μs. Let us be generous: 50ms.
216 delay(50);
217 hardReset = true;
218 }
219 }
220
221 if (!hardReset) { // Perform a soft reset if we haven't triggered a hard reset above.
222 PCD_Reset();
223 }
224
225 // Reset baud rates
226 PCD_WriteRegister(TxModeReg, 0x00);
227 PCD_WriteRegister(RxModeReg, 0x00);
228 // Reset ModWidthReg
229 PCD_WriteRegister(ModWidthReg, 0x26);
230
231 // When communicating with a PICC we need a timeout if something goes wrong.
232 // f_timer = 13.56 MHz / (2*TPreScaler+1) where TPreScaler = [TPrescaler_Hi:TPrescaler_Lo].
233 // TPrescaler_Hi are the four low bits in TModeReg. TPrescaler_Lo is TPrescalerReg.
234 PCD_WriteRegister(TModeReg, 0x80); // TAuto=1; timer starts automatically at the end of the transmission in all communication modes at all speeds
235 PCD_WriteRegister(TPrescalerReg, 0xA9); // TPreScaler = TModeReg[3..0]:TPrescalerReg, ie 0x0A9 = 169 => f_timer=40kHz, ie a timer period of 25μs.
236 PCD_WriteRegister(TReloadRegH, 0x03); // Reload timer with 0x3E8 = 1000, ie 25ms before timeout.
237 PCD_WriteRegister(TReloadRegL, 0xE8);
238
239 PCD_WriteRegister(TxASKReg, 0x40); // Default 0x00. Force a 100 % ASK modulation independent of the ModGsPReg register setting
240 PCD_WriteRegister(ModeReg, 0x3D); // Default 0x3F. Set the preset value for the CRC coprocessor for the CalcCRC command to 0x6363 (ISO 14443-3 part 6.2.4)
241 PCD_AntennaOn(); // Enable the antenna driver pins TX1 and TX2 (they were disabled by the reset)
242} // End PCD_Init()

267void MFRC522::PCD_Reset() {
268 PCD_WriteRegister(CommandReg, PCD_SoftReset); // Issue the SoftReset command.
269 // The datasheet does not mention how long the SoftRest command takes to complete.
270 // But the MFRC522 might have been in soft power-down mode (triggered by bit 4 of CommandReg)
271 // Section 8.8.2 in the datasheet says the oscillator start-up time is the start up time of the crystal + 37,74μs. Let us be generous: 50ms.
272 uint8_t count = 0;
273 do {
274 // Wait for the PowerDown bit in CommandReg to be cleared (max 3x50ms)
275 delay(50);
276 } while ((PCD_ReadRegister(CommandReg) & (1 << 4)) && (++count) < 3);
277} // End PCD_Reset()

326bool MFRC522::PCD_PerformSelfTest() {
327 // This follows directly the steps outlined in 16.1.1
328 // 1. Perform a soft reset.
329 PCD_Reset();
330
331 // 2. Clear the internal buffer by writing 25 bytes of 00h
332 byte ZEROES[25] = {0x00};
333 PCD_WriteRegister(FIFOLevelReg, 0x80); // flush the FIFO buffer
334 PCD_WriteRegister(FIFODataReg, 25, ZEROES); // write 25 bytes of 00h to FIFO
335 PCD_WriteRegister(CommandReg, PCD_Mem); // transfer to internal buffer
336
337 // 3. Enable self-test
338 PCD_WriteRegister(AutoTestReg, 0x09);
339
340 // 4. Write 00h to FIFO buffer
341 PCD_WriteRegister(FIFODataReg, 0x00);
342
343 // 5. Start self-test by issuing the CalcCRC command
344 PCD_WriteRegister(CommandReg, PCD_CalcCRC);
345
346 // 6. Wait for self-test to complete
347 byte n;
348 for (uint8_t i = 0; i < 0xFF; i++) {
349 // The datasheet does not specify exact completion condition except
350 // that FIFO buffer should contain 64 bytes.
351 // While selftest is initiated by CalcCRC command
352 // it behaves differently from normal CRC computation,
353 // so one can't reliably use DivIrqReg to check for completion.
354 // It is reported that some devices does not trigger CRCIRq flag
355 // during selftest.
356 n = PCD_ReadRegister(FIFOLevelReg);
357 if (n >= 64) {
358 break;
359 }
360 }
361 PCD_WriteRegister(CommandReg, PCD_Idle); // Stop calculating CRC for new content in the FIFO.
362
363 // 7. Read out resulting 64 bytes from the FIFO buffer.
364 byte result[64];
365 PCD_ReadRegister(FIFODataReg, 64, result, 0);
366
367 // Auto self-test done
368 // Reset AutoTestReg register to be 0 again. Required for normal operation.
369 PCD_WriteRegister(AutoTestReg, 0x00);
370
371 // Determine firmware version (see section 9.3.4.8 in spec)
372 byte version = PCD_ReadRegister(VersionReg);
373
374 // Pick the appropriate reference values
375 const byte *reference;
376 switch (version) {
377 case 0x88: // Fudan Semiconductor FM17522 clone
378 reference = FM17522_firmware_reference;
379 break;
380 case 0x90: // Version 0.0
381 reference = MFRC522_firmware_referenceV0_0;
382 break;
383 case 0x91: // Version 1.0
384 reference = MFRC522_firmware_referenceV1_0;
385 break;
386 case 0x92: // Version 2.0
387 reference = MFRC522_firmware_referenceV2_0;
388 break;
389 default: // Unknown version
390 return false; // abort test
391 }
392
393 // Verify that the results match up to our expectations
394 for (uint8_t i = 0; i < 64; i++) {
395 if (result[i] != pgm_read_byte(&(reference[i]))) {
396 return false;
397 }
398 }
399
400 // 8. Perform a re-init, because PCD does not work after test.
401 // Reset does not work as expected.
402 // "Auto self-test done" does not work as expected.
403 PCD_Init();
404
405 // Test passed; all is good.
406 return true;
407} // End PCD_PerformSelfTest()

467MFRC522::StatusCode MFRC522::PCD_CommunicateWithPICC( byte command,
468 byte waitIRq,
469 byte *sendData,
470 byte sendLen,
471 byte *backData,
472 byte *backLen,
473 byte *validBits,
474 byte rxAlign,
475 bool checkCRC
476 ) {
477 // Prepare values for BitFramingReg
478 byte txLastBits = validBits ? *validBits : 0;
479 byte bitFraming = (rxAlign << 4) + txLastBits; // RxAlign = BitFramingReg[6..4]. TxLastBits = BitFramingReg[2..0]
480
481 PCD_WriteRegister(CommandReg, PCD_Idle); // Stop any active command.
482 PCD_WriteRegister(ComIrqReg, 0x7F); // Clear all seven interrupt request bits
483 PCD_WriteRegister(FIFOLevelReg, 0x80); // FlushBuffer = 1, FIFO initialization
484 PCD_WriteRegister(FIFODataReg, sendLen, sendData); // Write sendData to the FIFO
485 PCD_WriteRegister(BitFramingReg, bitFraming); // Bit adjustments
486 PCD_WriteRegister(CommandReg, command); // Execute the command
487 if (command == PCD_Transceive) {
488 PCD_SetRegisterBitMask(BitFramingReg, 0x80); // StartSend=1, transmission of data starts
489 }
490
491 // In PCD_Init() we set the TAuto flag in TModeReg. This means the timer
492 // automatically starts when the PCD stops transmitting.
493 //
494 // Wait here for the command to complete. The bits specified in the
495 // `waitIRq` parameter define what bits constitute a completed command.
496 // When they are set in the ComIrqReg register, then the command is
497 // considered complete. If the command is not indicated as complete in
498 // ~36ms, then consider the command as timed out.
499 const uint32_t deadline = millis() + 36;
500 bool completed = false;
501
502 do {
503 byte n = PCD_ReadRegister(ComIrqReg); // ComIrqReg[7..0] bits are: Set1 TxIRq RxIRq IdleIRq HiAlertIRq LoAlertIRq ErrIRq TimerIRq
504 if (n & waitIRq) { // One of the interrupts that signal success has been set.
505 completed = true;
506 break;
507 }
508 if (n & 0x01) { // Timer interrupt - nothing received in 25ms
509 return STATUS_TIMEOUT;
510 }
511 yield();
512 }
513 while (static_cast<uint32_t> (millis()) < deadline);
514
515 // 36ms and nothing happened. Communication with the MFRC522 might be down.
516 if (!completed) {
517 return STATUS_TIMEOUT;
518 }
519
520 // Stop now if any errors except collisions were detected.
521 byte errorRegValue = PCD_ReadRegister(ErrorReg); // ErrorReg[7..0] bits are: WrErr TempErr reserved BufferOvfl CollErr CRCErr ParityErr ProtocolErr
522 if (errorRegValue & 0x13) { // BufferOvfl ParityErr ProtocolErr
523 return STATUS_ERROR;
524 }
525
526 byte _validBits = 0;
527
528 // If the caller wants data back, get it from the MFRC522.
529 if (backData && backLen) {
530 byte n = PCD_ReadRegister(FIFOLevelReg); // Number of bytes in the FIFO
531 if (n > *backLen) {
532 return STATUS_NO_ROOM;
533 }
534 *backLen = n; // Number of bytes returned
535 PCD_ReadRegister(FIFODataReg, n, backData, rxAlign); // Get received data from FIFO
536 _validBits = PCD_ReadRegister(ControlReg) & 0x07; // RxLastBits[2:0] indicates the number of valid bits in the last received byte. If this value is 000b, the whole byte is valid.
537 if (validBits) {
538 *validBits = _validBits;
539 }
540 }
541
542 // Tell about collisions
543 if (errorRegValue & 0x08) { // CollErr
544 return STATUS_COLLISION;
545 }
546
547 // Perform CRC_A validation if requested.
548 if (backData && backLen && checkCRC) {
549 // In this case a MIFARE Classic NAK is not OK.
550 if (*backLen == 1 && _validBits == 4) {
551 return STATUS_MIFARE_NACK;
552 }
553 // We need at least the CRC_A value and all 8 bits of the last byte must be received.
554 if (*backLen < 2 || _validBits != 0) {
555 return STATUS_CRC_WRONG;
556 }
557 // Verify CRC_A - do our own calculation and store the control in controlBuffer.
558 byte controlBuffer[2];
559 MFRC522::StatusCode status = PCD_CalculateCRC(&backData[0], *backLen - 2, &controlBuffer[0]);
560 if (status != STATUS_OK) {
561 return status;
562 }
563 if ((backData[*backLen - 2] != controlBuffer[0]) || (backData[*backLen - 1] != controlBuffer[1])) {
564 return STATUS_CRC_WRONG;
565 }
566 }
567
568 return STATUS_OK;
569} // End PCD_CommunicateWithPICC()

640MFRC522::StatusCode MFRC522::PICC_Select( Uid *uid,
641 byte validBits
642 ) {
643 bool uidComplete;
644 bool selectDone;
645 bool useCascadeTag;
646 byte cascadeLevel = 1;
647 MFRC522::StatusCode result;
648 byte count;
649 byte checkBit;
650 byte index;
651 byte uidIndex; // The first index in uid->uidByte[] that is used in the current Cascade Level.
652 int8_t currentLevelKnownBits; // The number of known UID bits in the current Cascade Level.
653 byte buffer[9]; // The SELECT/ANTICOLLISION commands uses a 7 byte standard frame + 2 bytes CRC_A
654 byte bufferUsed; // The number of bytes used in the buffer, ie the number of bytes to transfer to the FIFO.
655 byte rxAlign; // Used in BitFramingReg. Defines the bit position for the first bit received.
656 byte txLastBits; // Used in BitFramingReg. The number of valid bits in the last transmitted byte.
657 byte *responseBuffer;
658 byte responseLength;
659
660 // Description of buffer structure:
661 // Byte 0: SEL Indicates the Cascade Level: PICC_CMD_SEL_CL1, PICC_CMD_SEL_CL2 or PICC_CMD_SEL_CL3
662 // Byte 1: NVB Number of Valid Bits (in complete command, not just the UID): High nibble: complete bytes, Low nibble: Extra bits.
663 // Byte 2: UID-data or CT See explanation below. CT means Cascade Tag.
664 // Byte 3: UID-data
665 // Byte 4: UID-data
666 // Byte 5: UID-data
667 // Byte 6: BCC Block Check Character - XOR of bytes 2-5
668 // Byte 7: CRC_A
669 // Byte 8: CRC_A
670 // The BCC and CRC_A are only transmitted if we know all the UID bits of the current Cascade Level.
671 //
672 // Description of bytes 2-5: (Section 6.5.4 of the ISO/IEC 14443-3 draft: UID contents and cascade levels)
673 // UID size Cascade level Byte2 Byte3 Byte4 Byte5
674 // ======== ============= ===== ===== ===== =====
675 // 4 bytes 1 uid0 uid1 uid2 uid3
676 // 7 bytes 1 CT uid0 uid1 uid2
677 // 2 uid3 uid4 uid5 uid6
678 // 10 bytes 1 CT uid0 uid1 uid2
679 // 2 CT uid3 uid4 uid5
680 // 3 uid6 uid7 uid8 uid9
681
682 // Sanity checks
683 if (validBits > 80) {
684 return STATUS_INVALID;
685 }
686
687 // Prepare MFRC522
688 PCD_ClearRegisterBitMask(CollReg, 0x80); // ValuesAfterColl=1 => Bits received after collision are cleared.
689
690 // Repeat Cascade Level loop until we have a complete UID.
691 uidComplete = false;
692 while (!uidComplete) {
693 // Set the Cascade Level in the SEL byte, find out if we need to use the Cascade Tag in byte 2.
694 switch (cascadeLevel) {
695 case 1:
696 buffer[0] = PICC_CMD_SEL_CL1;
697 uidIndex = 0;
698 useCascadeTag = validBits && uid->size > 4; // When we know that the UID has more than 4 bytes
699 break;
700
701 case 2:
702 buffer[0] = PICC_CMD_SEL_CL2;
703 uidIndex = 3;
704 useCascadeTag = validBits && uid->size > 7; // When we know that the UID has more than 7 bytes
705 break;
706
707 case 3:
708 buffer[0] = PICC_CMD_SEL_CL3;
709 uidIndex = 6;
710 useCascadeTag = false; // Never used in CL3.
711 break;
712
713 default:
714 return STATUS_INTERNAL_ERROR;
715 break;
716 }
717
718 // How many UID bits are known in this Cascade Level?
719 currentLevelKnownBits = validBits - (8 * uidIndex);
720 if (currentLevelKnownBits < 0) {
721 currentLevelKnownBits = 0;
722 }
723 // Copy the known bits from uid->uidByte[] to buffer[]
724 index = 2; // destination index in buffer[]
725 if (useCascadeTag) {
726 buffer[index++] = PICC_CMD_CT;
727 }
728 byte bytesToCopy = currentLevelKnownBits / 8 + (currentLevelKnownBits % 8 ? 1 : 0); // The number of bytes needed to represent the known bits for this level.
729 if (bytesToCopy) {
730 byte maxBytes = useCascadeTag ? 3 : 4; // Max 4 bytes in each Cascade Level. Only 3 left if we use the Cascade Tag
731 if (bytesToCopy > maxBytes) {
732 bytesToCopy = maxBytes;
733 }
734 for (count = 0; count < bytesToCopy; count++) {
735 buffer[index++] = uid->uidByte[uidIndex + count];
736 }
737 }
738 // Now that the data has been copied we need to include the 8 bits in CT in currentLevelKnownBits
739 if (useCascadeTag) {
740 currentLevelKnownBits += 8;
741 }
742
743 // Repeat anti collision loop until we can transmit all UID bits + BCC and receive a SAK - max 32 iterations.
744 selectDone = false;
745 while (!selectDone) {
746 // Find out how many bits and bytes to send and receive.
747 if (currentLevelKnownBits >= 32) { // All UID bits in this Cascade Level are known. This is a SELECT.
748 //Serial.print(F("SELECT: currentLevelKnownBits=")); Serial.println(currentLevelKnownBits, DEC);
749 buffer[1] = 0x70; // NVB - Number of Valid Bits: Seven whole bytes
750 // Calculate BCC - Block Check Character
751 buffer[6] = buffer[2] ^ buffer[3] ^ buffer[4] ^ buffer[5];
752 // Calculate CRC_A
753 result = PCD_CalculateCRC(buffer, 7, &buffer[7]);
754 if (result != STATUS_OK) {
755 return result;
756 }
757 txLastBits = 0; // 0 => All 8 bits are valid.
758 bufferUsed = 9;
759 // Store response in the last 3 bytes of buffer (BCC and CRC_A - not needed after tx)
760 responseBuffer = &buffer[6];
761 responseLength = 3;
762 }
763 else { // This is an ANTICOLLISION.
764 //Serial.print(F("ANTICOLLISION: currentLevelKnownBits=")); Serial.println(currentLevelKnownBits, DEC);
765 txLastBits = currentLevelKnownBits % 8;
766 count = currentLevelKnownBits / 8; // Number of whole bytes in the UID part.
767 index = 2 + count; // Number of whole bytes: SEL + NVB + UIDs
768 buffer[1] = (index << 4) + txLastBits; // NVB - Number of Valid Bits
769 bufferUsed = index + (txLastBits ? 1 : 0);
770 // Store response in the unused part of buffer
771 responseBuffer = &buffer[index];
772 responseLength = sizeof(buffer) - index;
773 }
774
775 // Set bit adjustments
776 rxAlign = txLastBits; // Having a separate variable is overkill. But it makes the next line easier to read.
777 PCD_WriteRegister(BitFramingReg, (rxAlign << 4) + txLastBits); // RxAlign = BitFramingReg[6..4]. TxLastBits = BitFramingReg[2..0]
778
779 // Transmit the buffer and receive the response.
780 result = PCD_TransceiveData(buffer, bufferUsed, responseBuffer, &responseLength, &txLastBits, rxAlign);
781 if (result == STATUS_COLLISION) { // More than one PICC in the field => collision.
782 byte valueOfCollReg = PCD_ReadRegister(CollReg); // CollReg[7..0] bits are: ValuesAfterColl reserved CollPosNotValid CollPos[4:0]
783 if (valueOfCollReg & 0x20) { // CollPosNotValid
784 return STATUS_COLLISION; // Without a valid collision position we cannot continue
785 }
786 byte collisionPos = valueOfCollReg & 0x1F; // Values 0-31, 0 means bit 32.
787 if (collisionPos == 0) {
788 collisionPos = 32;
789 }
790 if (collisionPos <= currentLevelKnownBits) { // No progress - should not happen
791 return STATUS_INTERNAL_ERROR;
792 }
793 // Choose the PICC with the bit set.
794 currentLevelKnownBits = collisionPos;
795 count = currentLevelKnownBits % 8; // The bit to modify
796 checkBit = (currentLevelKnownBits - 1) % 8;
797 index = 1 + (currentLevelKnownBits / 8) + (count ? 1 : 0); // First byte is index 0.
798 buffer[index] |= (1 << checkBit);
799 }
800 else if (result != STATUS_OK) {
801 return result;
802 }
803 else { // STATUS_OK
804 if (currentLevelKnownBits >= 32) { // This was a SELECT.
805 selectDone = true; // No more anticollision
806 // We continue below outside the while.
807 }
808 else { // This was an ANTICOLLISION.
809 // We now have all 32 bits of the UID in this Cascade Level
810 currentLevelKnownBits = 32;
811 // Run loop again to do the SELECT.
812 }
813 }
814 } // End of while (!selectDone)
815
816 // We do not check the CBB - it was constructed by us above.
817
818 // Copy the found UID bytes from buffer[] to uid->uidByte[]
819 index = (buffer[2] == PICC_CMD_CT) ? 3 : 2; // source index in buffer[]
820 bytesToCopy = (buffer[2] == PICC_CMD_CT) ? 3 : 4;
821 for (count = 0; count < bytesToCopy; count++) {
822 uid->uidByte[uidIndex + count] = buffer[index++];
823 }
824
825 // Check response SAK (Select Acknowledge)
826 if (responseLength != 3 || txLastBits != 0) { // SAK must be exactly 24 bits (1 byte + CRC_A).
827 return STATUS_ERROR;
828 }
829 // Verify CRC_A - do our own calculation and store the control in buffer[2..3] - those bytes are not needed anymore.
830 result = PCD_CalculateCRC(responseBuffer, 1, &buffer[2]);
831 if (result != STATUS_OK) {
832 return result;
833 }
834 if ((buffer[2] != responseBuffer[1]) || (buffer[3] != responseBuffer[2])) {
835 return STATUS_CRC_WRONG;
836 }
837 if (responseBuffer[0] & 0x04) { // Cascade bit set - UID not complete yes
838 cascadeLevel++;
839 }
840 else {
841 uidComplete = true;
842 uid->sak = responseBuffer[0];
843 }
844 } // End of while (!uidComplete)
845
846 // Set correct uid->size
847 uid->size = 3 * cascadeLevel + 1;
848
849 return STATUS_OK;
850} // End PICC_Select()

1178 byte buffer[18];
1179
1180 // Translate the int32_t into 4 bytes; repeated 2x in value block
1181 buffer[0] = buffer[ 8] = (value & 0xFF);
1182 buffer[1] = buffer[ 9] = (value & 0xFF00) >> 8;
1183 buffer[2] = buffer[10] = (value & 0xFF0000) >> 16;
1184 buffer[3] = buffer[11] = (value & 0xFF000000) >> 24;

1186 buffer[4] = ~buffer[0];

1188 buffer[6] = ~buffer[2];
1189 buffer[7] = ~buffer[3];
1190 // Address 2x with inverse address 2x
1191 buffer[12] = buffer[14] = blockAddr;
1192 buffer[13] = buffer[15] = ~blockAddr;
1193
1194 // Write the whole data block
1195 return MIFARE_Write(blockAddr, buffer, 16);

1255 byte sendLen,
1256 bool acceptTimeout
1257 ) {

1509void MFRC522::PICC_DumpMifareClassicSectorToSerial(Uid *uid,
1510 MIFARE_Key *key,
1511 byte sector
1512 ) {
1513 MFRC522::StatusCode status;
1514 byte firstBlock; // Address of lowest address to dump actually last block dumped)
1515 byte no_of_blocks; // Number of blocks in sector
1516 bool isSectorTrailer; // Set to true while handling the "last" (ie highest address) in the sector.
1517
1518 // The access bits are stored in a peculiar fashion.
1519 // There are four groups:
1520 // g[3] Access bits for the sector trailer, block 3 (for sectors 0-31) or block 15 (for sectors 32-39)
1521 // g[2] Access bits for block 2 (for sectors 0-31) or blocks 10-14 (for sectors 32-39)
1522 // g[1] Access bits for block 1 (for sectors 0-31) or blocks 5-9 (for sectors 32-39)
1523 // g[0] Access bits for block 0 (for sectors 0-31) or blocks 0-4 (for sectors 32-39)
1524 // Each group has access bits [C1 C2 C3]. In this code C1 is MSB and C3 is LSB.
1525 // The four CX bits are stored together in a nible cx and an inverted nible cx_.
1526 byte c1, c2, c3; // Nibbles
1527 byte c1_, c2_, c3_; // Inverted nibbles
1528 bool invertedError; // True if one of the inverted nibbles did not match
1529 byte g[4]; // Access bits for each of the four groups.
1530 byte group; // 0-3 - active group for access bits
1531 bool firstInGroup; // True for the first block dumped in the group
1532
1533 // Determine position and size of sector.
1534 if (sector < 32) { // Sectors 0..31 has 4 blocks each
1535 no_of_blocks = 4;
1536 firstBlock = sector * no_of_blocks;
1537 }
1538 else if (sector < 40) { // Sectors 32-39 has 16 blocks each
1539 no_of_blocks = 16;
1540 firstBlock = 128 + (sector - 32) * no_of_blocks;
1541 }
1542 else { // Illegal input, no MIFARE Classic PICC has more than 40 sectors.
1543 return;
1544 }
1545
1546 // Dump blocks, highest address first.
1547 byte byteCount;
1548 byte buffer[18];
1549 byte blockAddr;
1550 isSectorTrailer = true;
1551 invertedError = false; // Avoid "unused variable" warning.
1552 for (int8_t blockOffset = no_of_blocks - 1; blockOffset >= 0; blockOffset--) {
1553 blockAddr = firstBlock + blockOffset;
1554 // Sector number - only on first line
1555 if (isSectorTrailer) {
1556 if(sector < 10)
1557 Serial.print(F(" ")); // Pad with spaces
1558 else
1559 Serial.print(F(" ")); // Pad with spaces
1560 Serial.print(sector);
1561 Serial.print(F(" "));
1562 }
1563 else {
1564 Serial.print(F(" "));
1565 }
1566 // Block number
1567 if(blockAddr < 10)
1568 Serial.print(F(" ")); // Pad with spaces
1569 else {
1570 if(blockAddr < 100)
1571 Serial.print(F(" ")); // Pad with spaces
1572 else
1573 Serial.print(F(" ")); // Pad with spaces
1574 }
1575 Serial.print(blockAddr);
1576 Serial.print(F(" "));
1577 // Establish encrypted communications before reading the first block
1578 if (isSectorTrailer) {
1579 status = PCD_Authenticate(PICC_CMD_MF_AUTH_KEY_A, firstBlock, key, uid);
1580 if (status != STATUS_OK) {
1581 Serial.print(F("PCD_Authenticate() failed: "));
1582 Serial.println(GetStatusCodeName(status));
1583 return;
1584 }
1585 }
1586 // Read block
1587 byteCount = sizeof(buffer);
1588 status = MIFARE_Read(blockAddr, buffer, &byteCount);
1589 if (status != STATUS_OK) {
1590 Serial.print(F("MIFARE_Read() failed: "));
1591 Serial.println(GetStatusCodeName(status));
1592 continue;
1593 }
1594 // Dump data
1595 for (byte index = 0; index < 16; index++) {
1596 if(buffer[index] < 0x10)
1597 Serial.print(F(" 0"));
1598 else
1599 Serial.print(F(" "));
1600 Serial.print(buffer[index], HEX);
1601 if ((index % 4) == 3) {
1602 Serial.print(F(" "));
1603 }
1604 }
1605 // Parse sector trailer data
1606 if (isSectorTrailer) {
1607 c1 = buffer[7] >> 4;
1608 c2 = buffer[8] & 0xF;
1609 c3 = buffer[8] >> 4;
1610 c1_ = buffer[6] & 0xF;
1611 c2_ = buffer[6] >> 4;
1612 c3_ = buffer[7] & 0xF;
1613 invertedError = (c1 != (~c1_ & 0xF)) || (c2 != (~c2_ & 0xF)) || (c3 != (~c3_ & 0xF));
1614 g[0] = ((c1 & 1) << 2) | ((c2 & 1) << 1) | ((c3 & 1) << 0);
1615 g[1] = ((c1 & 2) << 1) | ((c2 & 2) << 0) | ((c3 & 2) >> 1);
1616 g[2] = ((c1 & 4) << 0) | ((c2 & 4) >> 1) | ((c3 & 4) >> 2);
1617 g[3] = ((c1 & 8) >> 1) | ((c2 & 8) >> 2) | ((c3 & 8) >> 3);
1618 isSectorTrailer = false;
1619 }
1620
1621 // Which access group is this block in?
1622 if (no_of_blocks == 4) {
1623 group = blockOffset;
1624 firstInGroup = true;
1625 }
1626 else {
1627 group = blockOffset / 5;
1628 firstInGroup = (group == 3) || (group != (blockOffset + 1) / 5);
1629 }
1630
1631 if (firstInGroup) {
1632 // Print access bits
1633 Serial.print(F(" [ "));
1634 Serial.print((g[group] >> 2) & 1, DEC); Serial.print(F(" "));
1635 Serial.print((g[group] >> 1) & 1, DEC); Serial.print(F(" "));
1636 Serial.print((g[group] >> 0) & 1, DEC);
1637 Serial.print(F(" ] "));
1638 if (invertedError) {
1639 Serial.print(F(" Inverted access bits did not match! "));
1640 }
1641 }
1642
1643 if (group != 3 && (g[group] == 1 || g[group] == 6)) { // Not a sector trailer, a value block
1644 int32_t value = (int32_t(buffer[3])<<24) | (int32_t(buffer[2])<<16) | (int32_t(buffer[1])<<8) | int32_t(buffer[0]);
1645 Serial.print(F(" Value=0x")); Serial.print(value, HEX);
1646 Serial.print(F(" Adr=0x")); Serial.print(buffer[12], HEX);
1647 }
1648 Serial.println();
1649 }
1650
1651 return;
1652} // End PICC_DumpMifareClassicSectorToSerial()
