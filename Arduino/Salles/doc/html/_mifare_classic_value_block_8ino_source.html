1/**
2 * ----------------------------------------------------------------------------
3 * This is a MFRC522 library example; see https://github.com/miguelbalboa/rfid
4 * for further details and other examples.
5 *
6 * NOTE: The library file MFRC522.h has a lot of useful info. Please read it.
7 *
8 * Released into the public domain.
9 * ----------------------------------------------------------------------------
10 * This sample shows how to setup blocks on a MIFARE Classic PICC (= card/tag)
11 * to be in "Value Block" mode: in this mode the operations Increment/Decrement,
12 * Restore and Transfer can be used.
13 *
14 * BEWARE: Data will be written to the PICC, in sector #1 (blocks #4 to #7).
15 *
16 *
17 * Typical pin layout used:
18 * -----------------------------------------------------------------------------------------
19 * MFRC522 Arduino Arduino Arduino Arduino Arduino
20 * Reader/PCD Uno/101 Mega Nano v3 Leonardo/Micro Pro Micro
21 * Signal Pin Pin Pin Pin Pin Pin
22 * -----------------------------------------------------------------------------------------
23 * RST/Reset RST 9 5 D9 RESET/ICSP-5 RST
24 * SPI SS SDA(SS) 10 53 D10 10 10
25 * SPI MOSI MOSI 11 / ICSP-4 51 D11 ICSP-4 16
26 * SPI MISO MISO 12 / ICSP-1 50 D12 ICSP-1 14
27 * SPI SCK SCK 13 / ICSP-3 52 D13 ICSP-3 15
28 *
29 * More pin layouts for other boards can be found here: https://github.com/miguelbalboa/rfid#pin-layout
30 *
31 */

34#include <MFRC522.h>

36#define RST_PIN 9 // Configurable, see typical pin layout above
37#define SS_PIN 10 // Configurable, see typical pin layout above

39MFRC522 mfrc522(SS_PIN, RST_PIN); // Create MFRC522 instance.

44 * Initialize.

46void setup() {
47 Serial.begin(9600); // Initialize serial communications with the PC
48 while (!Serial); // Do nothing if no serial port is opened (added for Arduinos based on ATMEGA32U4)
49 SPI.begin(); // Init SPI bus
50 mfrc522.PCD_Init(); // Init MFRC522 card
51
52 // Prepare the key (used both as key A and as key B)
53 // using FFFFFFFFFFFFh which is the default at chip delivery from the factory
54 for (byte i = 0; i < 6; i++) {
55 key.keyByte[i] = 0xFF;
56 }
57
58 Serial.println(F("Scan a MIFARE Classic PICC to demonstrate Value Block mode."));
59 Serial.print(F("Using key (for A and B):"));
60 dump_byte_array(key.keyByte, MFRC522::MF_KEY_SIZE);
61 Serial.println();
62
63 Serial.println(F("BEWARE: Data will be written to the PICC, in sector #1"));
64}

66/**
67 * Main loop.
68 */
69void loop() {
70 // Reset the loop if no new card present on the sensor/reader. This saves the entire process when idle.
71 if ( ! mfrc522.PICC_IsNewCardPresent())
72 return;
73
74 // Select one of the cards
75 if ( ! mfrc522.PICC_ReadCardSerial())
76 return;
77
78 // Show some details of the PICC (that is: the tag/card)
79 Serial.print(F("Card UID:"));
80 dump_byte_array(mfrc522.uid.uidByte, mfrc522.uid.size);
81 Serial.println();
82 Serial.print(F("PICC type: "));
83 MFRC522::PICC_Type piccType = mfrc522.PICC_GetType(mfrc522.uid.sak);
84 Serial.println(mfrc522.PICC_GetTypeName(piccType));
85
86 // Check for compatibility
87 if ( piccType != MFRC522::PICC_TYPE_MIFARE_MINI
88 && piccType != MFRC522::PICC_TYPE_MIFARE_1K
89 && piccType != MFRC522::PICC_TYPE_MIFARE_4K) {
90 Serial.println(F("This sample only works with MIFARE Classic cards."));
91 return;
92 }
93
94 // In this sample we use the second sector,
95 // that is: sector #1, covering block #4 up to and including block #7
96 byte sector = 1;
97 byte valueBlockA = 5;
98 byte valueBlockB = 6;
99 byte trailerBlock = 7;
100 MFRC522::StatusCode status;
101 byte buffer[18];
102 byte size = sizeof(buffer);
103 int32_t value;
104
105 // Authenticate using key A
106 Serial.println(F("Authenticating using key A..."));
107 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_A, trailerBlock, &key, &(mfrc522.uid));
108 if (status != MFRC522::STATUS_OK) {
109 Serial.print(F("PCD_Authenticate() failed: "));
110 Serial.println(mfrc522.GetStatusCodeName(status));
111 return;
112 }
113
114 // Show the whole sector as it currently is
115 Serial.println(F("Current data in sector:"));
116 mfrc522.PICC_DumpMifareClassicSectorToSerial(&(mfrc522.uid), &key, sector);
117 Serial.println();
118
119 // We need a sector trailer that defines blocks 5 and 6 as Value Blocks and enables key B
120 // The last block in a sector (block #3 for Mifare Classic 1K) is the Sector Trailer.
121 // See http://www.nxp.com/documents/data_sheet/MF1S503x.pdf sections 8.6 and 8.7:
122 // Bytes 0-5: Key A
123 // Bytes 6-8: Access Bits
124 // Bytes 9: User data
125 // Bytes 10-15: Key B (or user data)
126 byte trailerBuffer[] = {
127 255, 255, 255, 255, 255, 255, // Keep default key A
128 0, 0, 0,
129 0,
130 255, 255, 255, 255, 255, 255}; // Keep default key B
131 // The access bits are stored in a peculiar fashion.
132 // There are four groups:
133 // g[0] Access bits for block 0 (for sectors 0-31)
134 // or blocks 0-4 (for sectors 32-39)
135 // g[1] Access bits for block 1 (for sectors 0-31)
136 // or blocks 5-9 (for sectors 32-39)
137 // g[2] Access bits for block 2 (for sectors 0-31)
138 // or blocks 10-14 (for sectors 32-39)
139 // g[3] Access bits for the Sector Trailer: block 3 (for sectors 0-31)
140 // or block 15 (for sectors 32-39)
141 // Each group has access bits [C1 C2 C3], in this code C1 is MSB and C3 is LSB.
142 // Determine the bit pattern needed using MIFARE_SetAccessBits:
143 // g0=0 access bits for block 0 (of this sector) using [0 0 0] = 000b = 0
144 // which means key A|B have r/w for block 0 of this sector
145 // which (in this example) translates to block #4 within sector #1;
146 // this is the transport configuration (at factory delivery).
147 // g1=6 access bits for block 1 (of this sector) using [1 1 0] = 110b = 6
148 // which means block 1 (of this sector) is used as a value block,
149 // which (in this example) translates to block #5 within sector #1;
150 // where key A|B have r, key B has w, key B can increment,
151 // and key A|B can decrement, transfer, and restore.
152 // g2=6 same thing for block 2 (of this sector): set it to a value block;
153 // which (in this example) translates to block #6 within sector #1;
154 // g3=3 access bits for block 3 (of this sector): the Sector Trailer here;
155 // using [0 1 1] = 011b = 3 which means only key B has r/w access
156 // to the Sector Trailer (block 3 of this sector) from now on
157 // which (in this example) translates to block #7 within sector #1;
158 mfrc522.MIFARE_SetAccessBits(&trailerBuffer[6], 0, 6, 6, 3);
159
160 // Read the sector trailer as it is currently stored on the PICC
161 Serial.println(F("Reading sector trailer..."));
162 status = mfrc522.MIFARE_Read(trailerBlock, buffer, &size);
163 if (status != MFRC522::STATUS_OK) {
164 Serial.print(F("MIFARE_Read() failed: "));
165 Serial.println(mfrc522.GetStatusCodeName(status));
166 return;
167 }
168 // Check if it matches the desired access pattern already;
169 // because if it does, we don't need to write it again...
170 if ( buffer[6] != trailerBuffer[6]
171 || buffer[7] != trailerBuffer[7]
172 || buffer[8] != trailerBuffer[8]) {
173 // They don't match (yet), so write it to the PICC
174 Serial.println(F("Writing new sector trailer..."));
175 status = mfrc522.MIFARE_Write(trailerBlock, trailerBuffer, 16);
176 if (status != MFRC522::STATUS_OK) {
177 Serial.print(F("MIFARE_Write() failed: "));
178 Serial.println(mfrc522.GetStatusCodeName(status));
179 return;
180 }
181 }
182
183 // Authenticate using key B
184 Serial.println(F("Authenticating again using key B..."));
185 status = mfrc522.PCD_Authenticate(MFRC522::PICC_CMD_MF_AUTH_KEY_B, trailerBlock, &key, &(mfrc522.uid));
186 if (status != MFRC522::STATUS_OK) {
187 Serial.print(F("PCD_Authenticate() failed: "));
188 Serial.println(mfrc522.GetStatusCodeName(status));
189 return;
190 }
191
192 // A value block has a 32 bit signed value stored three times
193 // and an 8 bit address stored 4 times. Make sure that valueBlockA
194 // and valueBlockB have that format (note that it will only format
195 // the block when it doesn't comply to the expected format already).
196 formatValueBlock(valueBlockA);
197 formatValueBlock(valueBlockB);
198
199 // Add 1 to the value of valueBlockA and store the result in valueBlockA.
200 Serial.print("Adding 1 to value of block "); Serial.println(valueBlockA);
201 status = mfrc522.MIFARE_Increment(valueBlockA, 1);
202 if (status != MFRC522::STATUS_OK) {
203 Serial.print(F("MIFARE_Increment() failed: "));
204 Serial.println(mfrc522.GetStatusCodeName(status));
205 return;
206 }
207 status = mfrc522.MIFARE_Transfer(valueBlockA);
208 if (status != MFRC522::STATUS_OK) {
209 Serial.print(F("MIFARE_Transfer() failed: "));
210 Serial.println(mfrc522.GetStatusCodeName(status));
211 return;
212 }
213 // Show the new value of valueBlockA
214 status = mfrc522.MIFARE_GetValue(valueBlockA, &value);
215 if (status != MFRC522::STATUS_OK) {
216 Serial.print(F("mifare_GetValue() failed: "));
217 Serial.println(mfrc522.GetStatusCodeName(status));
218 return;
219 }
220 Serial.print("New value of value block "); Serial.print(valueBlockA);
221 Serial.print(" = "); Serial.println(value);
222
223 // Decrement 10 from the value of valueBlockB and store the result in valueBlockB.
224 Serial.print("Subtracting 10 from value of block "); Serial.println(valueBlockB);
225 status = mfrc522.MIFARE_Decrement(valueBlockB, 10);
226 if (status != MFRC522::STATUS_OK) {
227 Serial.print(F("MIFARE_Decrement() failed: "));
228 Serial.println(mfrc522.GetStatusCodeName(status));
229 return;
230 }
231 status = mfrc522.MIFARE_Transfer(valueBlockB);
232 if (status != MFRC522::STATUS_OK) {
233 Serial.print(F("MIFARE_Transfer() failed: "));
234 Serial.println(mfrc522.GetStatusCodeName(status));
235 return;
236 }
237 // Show the new value of valueBlockB
238 status = mfrc522.MIFARE_GetValue(valueBlockB, &value);
239 if (status != MFRC522::STATUS_OK) {
240 Serial.print(F("mifare_GetValue() failed: "));
241 Serial.println(mfrc522.GetStatusCodeName(status));
242 return;
243 }
244 Serial.print(F("New value of value block ")); Serial.print(valueBlockB);
245 Serial.print(F(" = ")); Serial.println(value);
246 // Check some boundary...
247 if (value <= -100) {
248 Serial.println(F("Below -100, so resetting it to 255 = 0xFF just for fun..."));
249 status = mfrc522.MIFARE_SetValue(valueBlockB, 255);
250 if (status != MFRC522::STATUS_OK) {
251 Serial.print(F("mifare_SetValue() failed: "));
252 Serial.println(mfrc522.GetStatusCodeName(status));
253 return;
254 }
255 }
256
257 // Dump the sector data
258 mfrc522.PICC_DumpMifareClassicSectorToSerial(&(mfrc522.uid), &key, sector);
259 Serial.println();
260
261 // Halt PICC
262 mfrc522.PICC_HaltA();
263 // Stop encryption on PCD
264 mfrc522.PCD_StopCrypto1();
265}

267/**
268 * Helper routine to dump a byte array as hex values to Serial.
269 */
270void dump_byte_array(byte *buffer, byte bufferSize) {
271 for (byte i = 0; i < bufferSize; i++) {
272 Serial.print(buffer[i] < 0x10 ? " 0" : " ");
273 Serial.print(buffer[i], HEX);
274 }
275}

277/**
278 * Ensure that a given block is formatted as a Value Block.
279 */
280void formatValueBlock(byte blockAddr) {
281 byte buffer[18];
282 byte size = sizeof(buffer);
283 MFRC522::StatusCode status;
284
285 Serial.print(F("Reading block ")); Serial.println(blockAddr);
286 status = mfrc522.MIFARE_Read(blockAddr, buffer, &size);
287 if (status != MFRC522::STATUS_OK) {
288 Serial.print(F("MIFARE_Read() failed: "));
289 Serial.println(mfrc522.GetStatusCodeName(status));
290 return;
291 }
292
293 if ( (buffer[0] == (byte)~buffer[4])
294 && (buffer[1] == (byte)~buffer[5])
295 && (buffer[2] == (byte)~buffer[6])
296 && (buffer[3] == (byte)~buffer[7])
297
298 && (buffer[0] == buffer[8])
299 && (buffer[1] == buffer[9])
300 && (buffer[2] == buffer[10])
301 && (buffer[3] == buffer[11])
302
303 && (buffer[12] == (byte)~buffer[13])
304 && (buffer[12] == buffer[14])
305 && (buffer[12] == (byte)~buffer[15])) {
306 Serial.println(F("Block has correct Value Block format."));
307 }
308 else {
309 Serial.println(F("Formatting as Value Block..."));
310 byte valueBlock[] = {
311 0, 0, 0, 0,
312 255, 255, 255, 255,
313 0, 0, 0, 0,
314 blockAddr, ~blockAddr, blockAddr, ~blockAddr };
315 status = mfrc522.MIFARE_Write(blockAddr, valueBlock, 16);
316 if (status != MFRC522::STATUS_OK) {
317 Serial.print(F("MIFARE_Write() failed: "));
318 Serial.println(mfrc522.GetStatusCodeName(status));
319 }
320 }
